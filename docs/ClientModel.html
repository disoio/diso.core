    @find

  • ¶

    adapter to find model by _id

      @find : (args)->
    _id      = args._id
    callback = args.callback

    @get(
      message : {
        name : 'find'
        data : {
          model : @name 
          _id   : _id
        }
      }
      callback : (error, data)=>
        if error
          callback(error, null)
        else
          callback(null, data.model)
    )
